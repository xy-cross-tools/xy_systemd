```bash

xy_systemd -c project -n xy_systemd_demo
# 创建项目 [ xy_systemd_demo ] 成功!!!
# 项目路径 ==>>> /mnt/bs-media/Workspace/project/opensource/xy-web-service/xy_systemd/test/xy_systemd_demo

# >>>>>>>>>>>> xy_systemd - v1.0.1 <<<<<<<<<<<<<
#
# Hello World!!!
```
